\section{Ridge regression}

\begin{frame}{Regularization of the empirical risk}

    In the previous practical session, we studied, Ridge regression, a form \textbf{{regularization}} of the objective function for the linear regression problem.

    This enforces the unicity of the solution, even when $X^TX$ is not invertible ($X$ is the design matrix like before) at the cost of introducing a \textbf{{bias}} in the estimator. The unicity is guaranteed by the \textbf{{strong convexity}} of the new loss function (studies in the next exercises).
    
\end{frame}


\begin{frame}{Ridge regression estimator}
    We use the same notations as in the previous lectures.
    \begin{itemize}
	\item $X\in \mathbb{R}^{n,d}$
	\item $y\in \mathbb{R}^{n}$
	\item $\theta\in \mathbb{R}^d$
    \end{itemize}
\begin{equation}
    \label{risk:ridge}
    \hat{ \theta}_{\lambda} = \argmin_{\theta \in \mathbb{R}^d} \Big( \frac{1}{n} ||y-X\theta||_2^2 + \lambda || \theta||_2^2 \Big)
\end{equation}
with $\lambda >0$.
\end{frame}

\begin{frame}{Ridge regression estimator}
\begin{proposition}


    The Ridge regression estimator is unique even if $X^TX$ is not inversible and is given by
    \begin{equation*}
	\hat{ \theta}_{\lambda} = \frac{1}{n} ( \hat{\Sigma} +\lambda I_d)^{-1}X^Ty
    \end{equation*}
\end{proposition}
with 
    \begin{equation}
	\hat{\Sigma} = \frac{1}{n} X^TX\in \mathbb{R}^{d, d}
    \end{equation}
\end{frame}

\begin{frame}{Statistical analysis of ridge regression}

    We can compare the excess risk (difference between risk and Bayes risk) to that of OLS.

\begin{proposition}
    
    Under the linear model assumption, with fixed design setting, the ridge regression estimator has the following excess risk

    \begin{equation}
	E[ R( \hat{\theta}_{\lambda}]-R^* = \lambda^2\theta^{*T} ( \hat{\Sigma}+\lambda I_d)^{-2} \hat{\Sigma} \theta^*+ \frac{\sigma^2}{n} tr[ \hat{\Sigma}^2( \hat{\Sigma}+\lambda I_d)^{-2} ]
    \end{equation}
\end{proposition}

    
\end{frame}

\begin{frame}{Choice of $\lambda$}

    Is it possible that the excess risk is smaller with ridge regression than OLS ?

\begin{proposition}
    With the choice 

    \begin{equation}
	\label{eq:lambdaopt}
	\lambda^* = \frac{\sigma \sqrt{tr( \hat{\Sigma})}}{||\theta^*||_2 \sqrt{n} } 
    \end{equation}
    then

    \begin{equation}
	\label{ref:ridgeopt}
	E[ R( \hat{\theta}_{\lambda}]-R^*\leq \frac{\sigma \sqrt{tr( \hat{\Sigma})}||\theta^*||_2 }{ \sqrt{n} } 
    \end{equation}
\end{proposition}
\end{frame}

\begin{frame}{Choice of $\lambda$}
    \begin{itemize}
	\item[Ridge]
    \begin{equation}
	\label{ref:ridgeopt}
	E[ R( \hat{\theta}_{\lambda}]-R^*\leq \frac{\sigma \sqrt{tr( \hat{\Sigma})}||\theta^*||_2 }{ \sqrt{n} } 
    \end{equation}

\item[OLS]
    \begin{equation}
	\label{ref:ridgeopt}
	E[R(\hat{\theta}]-R^*  = \sigma^2 \frac{d}{n} 
    \end{equation}
    \end{itemize}

\begin{itemize}
    \item $ \frac{1}{n} $ (OLS) vs $ \frac{1}{ \sqrt{n} } $ (ridge), with different constants
    \item dimension-free bound for Ridge (maybe in the project)
\end{itemize}
\end{frame}

\begin{frame}{Hyperparameter search}
    \begin{itemize}
	\item In practical situations, the quantities involved in the computation of $\lambda^*$ in  \ref{eq:lambdaopt} are typically unknown. However this equation shows that there may \textbf{exist} a $\lambda$ with a better prediction performance than OLS, which can be found by cross validation in practice (previous and next TP).
	\item $\lambda$ is an example of \textbf{{hyperparameter}}. 
    \end{itemize}
\end{frame}

\section{Gradient algorithms}
\label{sec:gradients}

\begin{frame}{Context}

\end{frame}

\begin{frame}{Context}

    In machine learning, we often encounter problems in high dimension, where closed-form solutions are not avaible, or where even if they are available, the necessary computation time is too large.
    
    \textbf{{Example 1:}} Computing the OLS estimator requires a matrix inversion, which is $ \mathcal{O} (d^3)$.

    \begin{equation}
	\label{def:ols}
	\hat{ \theta}= (X^TX)^{-1}X^Ty
    \end{equation}

    High $d$ might be prohibitive for a numerical resolution.
\end{frame}

\begin{frame}{Context}

    In machine learning, we often encounter problems in high dimension, where closed-form solutions are not avaible, or where even if they are available, the necessary computation time is too large.
    
    \textbf{{Example 2:}} The cancellation of the gradient of the objective function with logistic loss has no closed-form solution.

\end{frame}

\begin{frame}{Context}


    Instead, we often use \textbf{{iterative}} algorithm such as Gradient descent (GD) or Stochastic gradient descent (SGD). SGD is the standard optimization algorithm for large-scale machine learning because (brutal summary):
    \begin{itemize}
    	\item SGD is roughly $n$ times faster than GD
	\item SGD's convergence is often satisfactory
    \end{itemize}
    
\end{frame}

\begin{frame}{Convergence speed}

    The properties and speed of convergence of GD and SGD are important properties that are extensively studied. They typically depend on:
    \begin{itemize}
	\item the convexity or strong convexity of the objective function $f$ (often the empirical risk).
	\item the regularity (smoothness constant) of the objective function $f$.
	    \item the statistical properties of the dataset
    \end{itemize}

    We will study some of these aspects during the practical sessions.
\end{frame}

\begin{frame}{Gradient descent}

    We want to minimize a function $f:\mathbb{R}^d \rightarrow \mathbb{R}$. \textbf{Gradient descent} iteratively performs the following update, until some criterion is met.

\begin{equation}
	\theta\leftarrow \theta-\gamma \nabla_f(\theta)
\end{equation}

$\gamma$ is called the \textbf{learning rate}, and is a very important hyperparameter. We will study some methods to tune it in practical sessions, like line seach.

\end{frame}

\begin{frame}{First-order method}

    GD is called a first-order method because it makes use of the first order derivative.

    The direction $-\nabla_{\theta} f(\theta)$ is the direction that minimizes the first order Taylor expansion of $f$ in $\theta$.

    \begin{equation}
	f(\theta+h) = f(\theta) + \langle \nabla_{\theta} f(\theta)|h \rangle + o(h)
    \end{equation}

\end{frame}

\begin{frame}{Stochastic gradient descent}
    In machine learning, we often consider an objective function of the form
    \begin{equation}
	f(\theta) = \frac{1}{n} \sum^{n}_{i=1} l(y_i, g_{\theta}(x_i))+\Omega(\theta)
    \end{equation}

    Example:
    \begin{itemize}
	\item $\Omega(\theta)= \lambda ||\theta||^2$.
	\item $l(y_i, g_{\theta}(x_i))=||y_i-\langle \theta|x_i \rangle||^2$
    \end{itemize}
\end{frame}

\begin{frame}{Batch gradient}
    In machine learning, we often consider an objective function of the form
    \begin{equation}
	f(\theta) = \frac{1}{n} \sum^{n}_{i=1} l(y_i, g_{\theta}(x_i))+\Omega(\theta)
    \end{equation}
    Gradient descent computes the \textbf{batch gradient} (also called \textbf{full gradient}) of $f$, which requires at least $n$ calculations, and each calculation also has a complexity that depends on the dimension $d$. When $n$ and $d$ are large, this might not be the best way to proceed.
\end{frame}

\begin{frame}{SGD}

    In machine learning, we often consider an objective function of the form
    \begin{equation}
	f(\theta) = \frac{1}{n} \sum^{n}_{i=1} l(y_i, g_{\theta}(x_i))+\Omega(\theta)
    \end{equation}

    SGD replaces the full gradient $\nabla_{\theta}f({\theta})$ by:

    \begin{equation}
	\nabla_{\theta}\Big[l(y_i, g_{\theta}(x_i))+\Omega(\theta)\Big](\theta)
    \end{equation}

    with $i$ being \textbf{randomly sampled} in $[1, n]$ (this is why it is called \textbf{stochastic}.)
\end{frame}
