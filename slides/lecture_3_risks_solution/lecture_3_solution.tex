\begin{frame}{}
    \exercice{}

    We observe the following dataset:
    \begin{equation}
        D_4 = \{(0, 1), (0, 0), (0, 0), (1, 0)\}
    \end{equation}

    Compute the empirical risks $R_4(f_1)$, $R_4(f_2)$, $R_4(f_3)$.

    \begin{equation*}
	R_n(f) = \frac{1}{n} \sum^{n}_{i=1} l(y_i, f(x_i))
    \end{equation*}
    
\end{frame}

\begin{frame}{}
    \begin{equation}
        D_4 = \{(0, 1), (0, 0), (0, 0), (1, 0)\}
    \end{equation}

    \begin{equation}
       \begin{aligned}
           \label{eq:}
           R_4(f_1) &= \frac{1}{4} \sum^{4}_{i=1} l(f_1(x_i), y_i)\\
           &= \frac{1}{4} \Big(l(f_1(0), 1) + l(f_1(0), 0))+ l(f_1(0), 0))+l(f_1(1), 0))  \Big)\\
           &= \frac{1}{4} \times 2\\
           &= \frac{1}{2} 
       \end{aligned} 
    \end{equation}
\end{frame}

\begin{frame}{}
    \exercice{} Consider the following random variable $(X,Y)$.
    \begin{itemize}
        \item 
            $ X\sim B( \frac{1}{2} )$, 
         $$
            Y = \left\{
                \begin{array}{ll}
                    B(p) \text{ if } X=1   \\
                    B(q) \text{ if } X=0   \\
                \end{array}
                \right.
                $$
\item $f_1: \{0, 1\} \rightarrow \{0, 1\} $:
         $$
            f = \left\{
                \begin{array}{ll}
                    1 \text{ if } x=1   \\
                    0 \text{ if } x=0   \\
                \end{array}
                \right.
                $$
    \end{itemize}

    \begin{equation}
        \begin{aligned}
            \label{eq:}
            R(f_1)&= E[l(Y, f(X))] \\
            &= 1\times P(Y\neq f(X))+ 0\times P(Y=f(X))\\
            &= P(Y\neq f(X))
        \end{aligned}
    \end{equation}
\end{frame}

\begin{frame}{}
    \begin{itemize}
        \item 
            $ X\sim B( \frac{1}{2} )$, 
         $$
            Y = \left\{
                \begin{array}{ll}
                    B(p) \text{ if } X=1   \\
                    B(q) \text{ if } X=0   \\
                \end{array}
                \right.
                $$
\item $f_1: \{0, 1\} \rightarrow \{0, 1\} $:
         $$
            f = \left\{
                \begin{array}{ll}

    \end{itemize}

    \begin{equation}

        \end{aligned}
    \end{equation}
\end{frame}

\begin{frame}{}
    \begin{equation}
        \begin{aligned}
            \label{eq:}
            R(f_1)&= E[l(Y, f(X))] \\
            &= 1\times P(Y\neq f(X))+ 0\times P(Y=f(X))\\
            &= P(Y\neq f(X))\\
            &= P((Y\neq f(X))\cap (X=1))+P((Y\neq f(X))\cap (X=0))\\

    \end{equation}
\end{frame}

\begin{frame}{}
    \begin{equation}
        \begin{aligned}
            \label{eq:}
            R(f_1)&= E[l(Y, f(X))] \\
            &= 1\times P(Y\neq f(X))+ 0\times P(Y=f(X))\\
            &= P(Y\neq f(X))\\
            &= P((Y\neq f(X))\cap (X=1))+P((Y\neq f(X))\cap (X=0))\\
            &= P((Y\neq f(X))| X=1)P(X=1)\\
            &+P((Y\neq f(X))| X=0)P(X=0)\\

        \end{aligned}
    \end{equation}
\end{frame}

\begin{frame}{}
    \begin{equation}
        \begin{aligned}
            \label{eq:}
            R(f_1)&= E[l(Y, f(X))] \\
            &= 1\times P(Y\neq f(X))+ 0\times P(Y=f(X))\\
