\section{Dimensionality reduction}%
\frame{\tableofcontents[currentsection]}
\label{sub:dimensionality_reduction}

\begin{frame}{Dimensionality reduction}
    We consider the space $ \mathcal{X} $ that contains the data, for either supervised or unsupervised learning. In machine learning, we often have $ \mathcal{X} \in \mathbb{R}^d$.
    \begin{itemize}
	\item If $d$ is large (e.g. $\geq 10^4$), the algorithms that run on the data might become too slow to be used, as their algorithmic complexity depends on $d$ (potentially in a quadratic or exponential way, curse of dimensionality)
    \end{itemize}
\end{frame}

\begin{frame}{Dimensionality reduction}
    We consider the space $ \mathcal{X} $ that contains the data, for either supervised or unsupervised learning. In machine learning, we often have $ \mathcal{X} \in \mathbb{R}^d$.
    \begin{itemize}
	\item If $d$ is large (e.g. $\geq 10^4$), the algorithms that run on the data might become too slow to be used, as their algorithmic complexity depends on $d$ (potentially in a quadratic or exponential way, curse of dimensionality)
	\item \textbf{{However}}, often the data might actually occupy a \textbf{{subspace}} of lower dimension $q$, or it may be possible to project the data on such a subspace without loosing too much information.
	    \begin{itemize}
		\item Working in a subspace of lower dimension might speed up the algorithms.
		    \item It may also allow visualization of the data.
	    \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Main methods of dimensionality reduction}
    \begin{itemize}
        \item \textbf{{feature selection}}: selecting a subset of the original dimensions.
	\item \textbf{{feature extraction}}: computing new features from the original features.
    \end{itemize}
\end{frame}

\subsection{Principal component analysis}%
\label{ssub:principal_component_analysis}

\begin{frame}{Principal component analysis (PCA)}
    \begin{itemize}
        \item PCA is a \textbf{{linear feature extraction}} technique.
	\item Points in $ \mathbb{R}^d$ are linearly projected on a well chosen affine subspace of $ \mathbb{R}^q$, with $q\leq d$.
    \end{itemize}
\end{frame}

\begin{frame}{Formalization as a (empirical) variance maximimisation problem}

Without loss of generality, we assume the data are \textbf{{centered}}, which means that

\begin{equation}
\bar{x} = \sum^{n}_{i=1} x_n=0\in \mathbb{R}^d
\end{equation}

$X$ is the design matrix like in OLS. The \textbf{{first principal componenent}} is a vector $w\in \mathbb{R}^d$, with $||w||=1$, such that $ \hat{Var}(w^Tx)$ is maximal, where $ \hat{Var}$ denotes the empirical variance.
\end{frame}


\begin{frame}{Variance}
    
    \begin{equation}
            \label{eq:}
            \overline{w^Tx} = w^T \overline{x} = 0
    \end{equation}

    Hence,

    \begin{equation}
        \begin{aligned}
            \label{eq:}
	    \hat{Var}(w^Tx) &= \frac{1}{n-1}  \sum^{n}_{i=1} \big((w^Tx)_i- \overline{w^Tx}  \big)^2\\
	    &= \frac{1}{n-1}  \sum^{n}_{i=1} \big(w^Tx_i \big)^2\\
        \end{aligned}
    \end{equation}
\end{frame}

\begin{frame}{Variance maximisation problem}

    We can then formulate the problem as finding $w$, $||w||=1$ such that

    \begin{equation}
	\sum^{n}_{i=1} \big(w^Tx_i \big)^2
    \end{equation}
     is maximal.
    
\end{frame}

\begin{frame}{First principal component}
    We look for $w$, $||w||=1$ such that
    \begin{equation}
	\sum^{n}_{i=1} \big(w^Tx_i \big)^2
    \end{equation}
     is maximal.
    \begin{proposition}
	$w$ is the eigenvector of $X^TX$ with largest eigenvalue $\lambda_{max}$.
    \end{proposition}
\end{frame}

\begin{frame}{First principal component}
    We look for $w$, $||w||=1$ such that
    \begin{equation}
	\sum^{n}_{i=1} \big(w^Tx_i \big)^2
    \end{equation}
     is maximal.
    \begin{proposition}
	\label{prop:pca}
	$w$ is the eigenvector of $X^TX$ with largest eigenvalue $\lambda_{max}$.
    \end{proposition}

    \exercice{\textbf{}} Show the proposition.
\end{frame}

\begin{frame}{Reconstruction error}
    Alternately, we can formulate the problem as a \textbf{{reconstruction error minimization.}} 

    \begin{equation}
	\mathbb{R}^d = \text{Vect}(w) \bigoplus\text{Vect}(w)^{\perp}
    \end{equation}

    and if $||w||=1$,
    
    \begin{equation}
	\begin{aligned}
	    \label{eq:}
	    \forall x\in \mathbb{R}^d, ||x||^2 &= ||(x^Tw)w||^2+||x-(x^Tw)w||^2\\
	    &= (x^Tw)^2+||x-(x^Tw)w||^2\\
	\end{aligned}
    \end{equation}
    
\end{frame}


\begin{frame}{Reconstruction error}

We can formulate the problem as a \textbf{{reconstruction error minimization.}} If $||w||=1$, 

    \begin{equation}
	\begin{aligned}
	    \label{eq:}
	    \forall x\in \mathbb{R}^d, ||x||^2 &= ||(x^Tw)w||^2+||x-(x^Tw)w||^2\\
	    &= (x^Tw)^2+||x-(x^Tw)w||^2\\
	\end{aligned}
    \end{equation}

    Hence,

    \begin{equation}
        \begin{aligned}
            \label{eq:}
	    \sum^{n}_{i=1} ||x_i||^2 &= \sum^{n}_{i=1} (x_i^Tw)^2+\sum^{n}_{i=1}||x_i-(x_i^Tw)w||^2\\
	    &= \hat{Var}(w^Tx) + \sum^{n}_{i=1}||x_i-(x_i^Tw)w||^2\\
        \end{aligned}
    \end{equation}
\end{frame}
    
\begin{frame}{Reconstruction error}

    \begin{equation}
	\sum^{n}_{i=1} ||x_i||^2 = \hat{Var}(w^Tx) + \sum^{n}_{i=1}||x_i-(x_i^Tw)w||^2
    \end{equation}

    We can see $\sum^{n}_{i=1}||x_i-(x_i^Tw)w||^2$ as a \textbf{{reconstruction error}}, when the data are projected on $\text{Vect}(w)$. 

    Maximizing the variance of the projections is equivalent to minimizing the reconstruction errors obtained by projection.
\end{frame}

\begin{frame}{Several principal components}

    Most of the time, we project the data on several principal components.

    \begin{itemize}
	\item 1] compute the first principal component $w_1$
	\item 2] project the data on $\text{Vect}(w_1)^{\perp}$
	\item 3] start again on the projected data
    \end{itemize}
\end{frame}

\begin{frame}{Reconstruction error}

    The interpretation stays the same. If $p_F(x)$ is the projection of $x$ on the subspace spanned by the principal components:

    \begin{equation}
	||x||^2 = ||p_F(x)||^2+||x-p_F(x)||^2
    \end{equation}

    The principal components are the largest eigenvectors of $X^TX\in \mathbb{R}^d, d$ with norm $1$. They are \textbf{{orthogonal}} to each other.
    
\end{frame}

\begin{frame}{Inertia}

    We can define an inertia:

    \begin{equation}
	I_F = \sum^{n}_{i=1} ||x-p_F(x)||^2
    \end{equation}
    
    We lool for the subspace that minimizes the inertia $I_F$.
\end{frame}

\begin{frame}{Explained variance}
    A natural question is: what is a relevant number of principal components ?

    A common quantity that is used is \textbf{{explained variance}}.  Each component $w_k$ carries a percentage of the total variance of the data.

    \begin{equation}
	\frac{ \hat{Var}(w_k^Tx)}{ \sum^{d}_{j=1} \hat{Var}(x^j)}
    \end{equation}
    where $\hat{Var}(x^j)$ is the variance of the (centered) feature $j$.

\begin{equation}
    \hat{Var}(x^j) = \frac{1}{n-1}\sum^{n}_{i=1} (x_i^j)^2
\end{equation}
\end{frame}

\section{Clustering}

\begin{frame}{Unsupervised learning}
 From a number of samples ${x_i}$, you want to retrieve information on their structure: \textbf{{modelisation}}. The three main unsupervised learning problems are:
    \begin{itemize}
        \item clustering
	\item density estimation
	\item dimensionality reduction
    \end{itemize}
\end{frame}

\begin{frame}{Clustering}
   \textbf{{Clustering}}  consists in partitioning the data. $\forall i, x_i\in \mathcal{X}^n$.
    \begin{equation}
	D_n = \{(x_i)_{i\in [1, \dots, n]}\}
    \end{equation}

    A \textbf{{partition}} is a set of $K$ subsets $A_k\subset D_n$, such that
    \begin{itemize}
        \item
	    \begin{equation}
		\cup_{k\in [1, \dots, K]}A_k = D_n
	    \end{equation}
	\item 
	    \begin{equation}
		\forall k\neq k', A_k\cap A_{k'} = \emptyset
	    \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}{Partitions}

    \begin{itemize}
	\item \textbf{{Example 1:}} $A$ is the set of even integers, $B$ the set ot odd integers. Is $(A, B)$ a partition of $ \mathbb{N} $ ?
	\item \textbf{{Example 2:}} $C$ is the set of multiples of $2$, $D$ the set of multiples of $3$. Is $(C, D)$ a partition of $ \mathbb{N} $ ?
    \end{itemize}
    
\end{frame}

\begin{frame}{Vector quantization}

    \textbf{{Vector quantization}} consists in computing \textbf{{prototypes}} $\Omega=(\omega_k)_{k\in [1, \dots, K]}\in \mathcal{X}^K$ that represent the data well.

    This implies that a \textbf{{metric}} is defined on $ \mathcal{X} $.

    Most often, this is interesting / useful if $ K<< n$.
\end{frame}

\begin{frame}{Voronoï subsets}

    We assume a loss (we can think of it as a distance here) $L$ is defined on $ \mathcal{X} $. The \textbf{Voronoï subset} of $\omega$ is defined as
    
    \begin{equation}
	V(\omega) = \{x\in D_n, \argmin_{\omega'\in \Omega}L(\omega', x)=\omega \}
    \end{equation}
    \begin{itemize}
        \item We assume that $ \argmin$ returns one single element.
	\item The Voronoï subsets form a partition of $D_n$.
    \end{itemize}
\end{frame}

\begin{frame}{Distortion}
    To measure the quality of a Voronoï partition, we introduce the \textbf{{distortion}} $R(\Omega)$.

    For each $x$, we note $h_{\Omega}(x)= \argmin_{\omega'\in \Omega}L(\omega', x)$.

    \begin{equation}
        \begin{aligned}
            \label{eq:}
	    R(\Omega)&= \frac{1}{n} \sum^{n}_{i=1}  L(x_i, h_{\Omega}(x_i))
        \end{aligned}
    \end{equation}
\end{frame}

\begin{frame}{Distortion}
    For each $x$, we note $h_{\Omega}(x)= \argmin_{\omega'\in \Omega}L(\omega', x)$.

    \begin{equation}
        \begin{aligned}
            \label{eq:}
	    R(\Omega)&= \frac{1}{n} \sum^{n}_{i=1}  L(x_i, h_{\Omega}(x_i))\\
	    &= \frac{1}{n} \sum_{\omega\in \Omega} \sum_{x\in V(\omega)}  L(x_i, h_{\Omega}(x_i))\\
	    &= \frac{1}{n} \sum_{\omega\in \Omega} V_{\Omega}(\omega)
        \end{aligned}
    \end{equation}

    with

    \begin{equation}
        V_{\Omega}(\omega)= \sum_{x\in V(\omega)}  L(x_i, h_{\Omega}(x_i))
    \end{equation}
\end{frame}

\begin{frame}{Minimum of distortion}

    We want to find the prototypes for which the distortion is \textbf{{minimal}}. 

    \begin{itemize}
        \item The set of prototypes minimizing distorsion might not be unique.
	\item We need to tune $K$ (number of prototypes).
    \end{itemize}
\end{frame}

\begin{frame}{Vector quantization techniques}
\begin{itemize}
    \item K-means
    \item Growing neural gas (GNG)
    \item Self-organizing maps
\end{itemize}
\end{frame}

\begin{frame}{K-means clustering}
    \begin{itemize}
        \item $\mathcal{X}  = \mathbb{R}^d$.
	\item $ L(x, y) = ||x-y||^2$.
    \end{itemize}
\end{frame}

\begin{frame}{Objective function}

With
\begin{itemize}
    \item $\Omega = \{\omega_1, \dots, \omega_K\}\in \mathbb{R}^{K, d}$.
    \item $z_i^k=1 $ if $x_i$ is assigned to $ \omega_k$, $z_i^k=0$ otherwise. $z= (z_i^k)\in \mathbb{R}^{n, K}$.
\end{itemize}
we define the objective function

\begin{equation}
    J(\Omega, z) = \sum^{n}_{i=1} \sum^{K}_{k=1} z_i^k||x_i-\omega_k||^2
\end{equation}

It is also called the \textbf{{inertia.}} 
    
\end{frame}

\begin{frame}{K-means algorithm}

\begin{algorithm}[H]
	\KwResult{$\Omega \in \mathbb{R}^{K, d}$}
	$\Omega\leftarrow $ \text{Random initialization}\;
	$z = M\text{ where } M \text{ is the } n\times K \text{ matrix with 0's}$\;
	\While{$\text{Convergence criteria is not satisfied}$}
 	{
	    $\text{a] Greedily minimize }J \text{ with respect to }z$\;
	    $\text{b] Minimize }J \text{ with respect to }\Omega$\;
  	}
  	return $\Omega$
	\caption{K-means (Loyd algorithm)}
\end{algorithm}
\end{frame}

\begin{frame}{Stopping criterion}
    
    To stop the algorithm, the norm of the difference between $\Omega_t$ and $\Omega_{t+1}$ must be smaller than a given tolerance. (e.g. $1e^{-4}$). Here it is a norm between matrix (Frobenius norm):

    \begin{equation}
	||A||_F = \sqrt{ \sum^{n}_{i=1} A_{ij}^2} 
    \end{equation}
    
\end{frame}

\begin{frame}{Minimization}

    We focus on step b]. How can we minimize $J$ with respect to $\Omega$ ?
    
\end{frame}

\begin{frame}{Minimization}

    \exercice{\textbf{Convexity:}}

    Show that $J(\Omega, z)$ is convex with respect to $ \Omega$.
\begin{equation}
    J(\Omega, z) = \sum^{n}_{i=1} \sum^{K}_{k=1} z_i^k||x_i-\omega_k||^2
\end{equation}
Hence, to minimize $J$ with respect to $\Omega$, we just need to cancel the gradient.
    
\end{frame}

\begin{frame}{Minimization}

    \exercice{\textbf{Gradient:}}
    \addtocounter{numexos}{-1}

    Compute the gradient of $J(\Omega, z)$ with respect to $\Omega$ and deduce the minimizer $\Omega^*$.
    \begin{itemize}
        \item $z$ is fixed
	\item we can see $\Omega$ has a vector of $ \mathbb{R}^{Kd}$.
    \end{itemize}
\end{frame}

\begin{frame}{Convexity}

    Is $J(\Omega, z)$ convex in $z$ ?
    
\end{frame}

\begin{frame}{Convexity}

    Is $J(\Omega, z)$ convex in $z$ ?

    \textbf{{No}}, as $z$ is not even defined on a convex set, so convexity can not be defined anyways !

    Hence, the function might have \textbf{{local minima}}.
    
\end{frame}

\begin{frame}{Random initialization}

    Hence, the result of K-means strongly depends on the initial position of the centroids.

    A common approach is to restart the algorithm several times and select the result with lowest inertia.
