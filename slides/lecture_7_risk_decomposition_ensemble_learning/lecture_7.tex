\section{Risks and risk decompositions}

\subsection{Bias, variance and optimization error}

\begin{frame}{Notion of risk decomposition}

    Context: usual supervised learning.
    \begin{itemize}
	\item empirical risk of estimator $f$: $R_n(f)$ (random variable)
	\item risk (real risk, generalization error) of estimator $f$: $R(f)$ (real number).
	    \item dataset $D_n$.
    \end{itemize}
    
We would like to interpret the risk of any estimator.

\end{frame}

\begin{frame}{Convergence of empirical risk}

    We fix $f\in H$ (hypothesis space).  We assume that the samples $(x_i,
    y_i)$ are i.i.d, with the distribution of $(X,Y)$, noted $\rho$. Then, according to the law of large numbers,
    under some assumptions (for instance, if the empirical risks are bounded),
    we have proven that in probability:

    \begin{equation}
        \lim_{n\rightarrow+\infty} R_n(f) = R(f)
    \end{equation}

    The empirical risk of a fixed $f$ converges to its real risk.
    
\end{frame}

\begin{frame}{Notion of risk decomposition}

    We would like to compare the real risk $R(g)$, for some estimator $g$, to $R^*=R(f^*)$, $f^*$ being the Bayes estimator.

\end{frame}

\begin{frame}{First decomposition}
\begin{itemize}
    \item $f^*$: Bayes predictor
    \item $F$: Hypothesis space
    \item $g$: some predictor ($\in F$).
\end{itemize}

\begin{equation}
    R(g)-R^* = \Big(R(g)- \inf_{f\in F}R(f)\Big) +\Big(\inf_{f\in F}R(f)-R^* \Big)
\end{equation}
\end{frame}

\begin{frame}{First decomposition}
\begin{itemize}
    \item $f^*$: Bayes predictor
    \item $F$: Hypothesis space
    \item $g$: some predictor
	\item $f_n$: empirical risk minimizer
\end{itemize}

    Most of the time, we will be interested in such a decomposition for the learned estimator, e.g. the empirical risk minimizer $f_n$. Now, $R(f_n)$ is a \textbf{random variable} !

\begin{equation}
    E[R(f_n)]-R^* = \Big(E[R(f_n)]- \inf_{f\in F}R(f)\Big) +\Big(\inf_{f\in F}R(f)-R^* \Big)
\end{equation}
\end{frame}

\begin{frame}{Underfitting and overfitting}

\textbf{{Approximation error (bias)}}: depends on $f^*$ and $F$, not on $ f_n$, $D_n$.
	\begin{equation*}
	    \inf_{f\in F}R(f)-R^*\geq 0
	\end{equation*}

\textbf{{Estimation error (variance, fluctuation error, stochastic error)}}: depends on $D_n$, $F$, $ f_n$. 
	\begin{equation*}
	    E(R( f_n))- \inf_{f\in F}R(f)\geq 0
	\end{equation*}

	\begin{itemize}
	    \item too small $F$ (compared to $f^*$): underfitting (large bias, small variance)
	    \item too large $F$ (compared to $n$): overffitting (small bias, large variance)
	\end{itemize}
\end{frame}


\begin{frame}{Deterministic bound on the estimation error}
We consider the best estimator in the hypothesis space $F$.
    \begin{equation*}
	f_a = \argmin_{h\in F}R(h)
    \end{equation*}

    \exercice{} Show that
    \begin{equation}
	R( f_n)-R(f_a) \leq 2 \sup_{h\in F}|R(h)-R_n(h)|
    \end{equation}
\end{frame}

\begin{frame}{Deterministic bound on the estimation error}

    \begin{equation}
	R( f_n)-R(f_a)\leq 2 \sup_{h\in F}|R(h)-R_n(h)|
    \end{equation}

   Later in the course, based on \textbf{{concentration inequalities }} we will further build on this result and prove a probabilistic bound of the form

    \begin{equation}
	R( f_n)-R(f_a)\leq \frac{C}{ \sqrt{n} } 
    \end{equation}

    (remember that by definition $0 \leq R( f_n)-R(f_a)$)

\end{frame}

\begin{frame}{Approximate solution}

    \begin{itemize}
        \item In machine learning, it is often not necessary to find the actual
            minimizer of the empirical risk, as there is an estimation error of $ \mathcal{O} ( \frac{1}{ \sqrt{n} }) $. \cite[]{Bottou2009}
        \item We can use an approximate solution $ \hat{f}_n$, such that 
            \begin{equation}
                R_n( \hat{f}_n)\leq R_n(f_n)+\rho
            \end{equation}
            with $\rho$ a predefined tolerance.
        \item This important because in large-scale ML, the \textbf{{computation time}} needs to be optimized.
    \end{itemize}
\end{frame}

\begin{frame}{Approximate solution}

    This gives a new risk decomposition:

\begin{equation}
    \begin{aligned}
        \label{eq:}
        E[R( \hat{f}_n)]-R^* &= \Big( E[R( \hat{f}_n)]-E[R(f_n)]\Big)\\
        &+ \Big(E[R(f_n)]- \inf_{f\in F}R(f)\Big)\\
        &+\Big(\inf_{f\in F}R(f)-R^* \Big)
    \end{aligned}
\end{equation}
    
\end{frame}

\begin{frame}

\begin{equation}
    \begin{aligned}
        \label{eq:}
        E[R( \hat{f}_n)]-R^* &= \Big( E[R( \hat{f}_n)]-E[R(f_n)]\Big)\\
        &+ \Big(E[R( f_n)]- \inf_{f\in F}R(f)\Big)\\
        &+\Big(\inf_{f\in F}R(f)-R^* \Big)
    \end{aligned}
\end{equation}

$E[R( \hat{f}_n)]-E[R(\tilde{f}_n)] $ is the \textbf{{optimization error.}} 

To conclude, we have:
\begin{itemize}
    \item an approximation error
    \item an estimation error
    \item an optimization error
\end{itemize}
    
\end{frame}

\subsection{Bounding the estimation error}

\begin{frame}{Deterministic bound on the estimation error}

    We will now focus on the estimation error

	\begin{equation*}
	    E\Big[R( f_n)\Big]- \inf_{f\in F}R(f)\geq 0
	\end{equation*}

	    $f_n$ is the empirical risk minimizer

We consider the best estimator in hypothesis space
    \begin{equation*}
	f_a = \argmin_{h\in F}R(h)
    \end{equation*}

    We have seen that
    \begin{equation}
	R( f_n)-R(f_a) \leq 2 \sup_{h\in F}|R(h)-R_n(h)|
    \end{equation}

\end{frame}

\begin{frame}{Deterministic bound on the estimation error}
    \begin{equation*}
	f_a = \argmin_{h\in F}R(h)
    \end{equation*}

    \begin{equation}
	f_n = \argmin_{h\in F}R_n(h)
    \end{equation}

    \begin{equation}
        \begin{aligned}
	    R(f_n)-R(f_a) &= \big(R(f_n)-R_n( f_n)\big)\\
	    &+\big(R_n( f_n)- R_n(f_a)\big)\\
	    &+\big(R_n(f_a)-R(f_a)\big)
        \end{aligned}
    \end{equation}

    But by definition $ f_n$ minimizes $R_n$, so $ \big( R_n ( f_n)-R_n(f_a)\big)\leq 0$.

    Finally:
    \begin{equation}
	R( f_n)-R(f_a)\leq 2 \sup_{h\in F}|R(h)-R_n(h)|
    \end{equation}

\end{frame}

\begin{frame}{Bound on the estimation error}
    If we are able to bound $\sup_{h\in F}|R(h)-R_n(h)|$, then we have a bound on the estimation error.

\end{frame}


\begin{frame}{Deterministic bound on the estimation error}
\begin{theorem}{Weak law of large numbers}

    Let $(X_i)_{i\in \mathbb{N} }$ be a sequence of i.i.d. variables that have a moment of order $2$. We note $m$ their expected value. Then

    \begin{equation}
	\forall \epsilon > 0, \lim_{n\rightarrow+\infty}P\big( |\frac{1}{n} \sum^{n}_{i=1} X_i-m|\geq \epsilon \big)= 0
    \end{equation}

We say that we have \textbf{{convergence in probability}}.
\end{theorem}

\textbf{{However}}, this is only an asymptotical result: it is a limit for $ n \rightarrow +\infty$.
    
\end{frame}


\begin{frame}{Bound on the estimation error}

    If we are able to bound $\sup_{h\in F}|R(h)-R_n(h)|$, then we have a bound on the estimation error.

    To do this, we will use some other mathematical results:

    \begin{itemize}
        \item Boole's inequality
	\item Hoeffding's inequality (non-asymptotical probabilistic bound)
    \end{itemize}
    
\end{frame}

\begin{frame}{Boole's inequality}
\begin{proposition}

    Let $A_1, A_2, \dots, $ be  acountable set of events of a probability space $\{\Omega, \mathcal{F}, P)$.

    Then, 

    \begin{equation}
	P\Big( \cup_{i\geq 1}A_i \Big) \leq \sum_{i\geq 1} P(A_i)
    \end{equation}
    
\end{proposition}


    This set might be infinite.
    
\end{frame}

\begin{frame}{Boole's inequality}
\begin{proposition}

    Let $A_1, A_2, \dots, $ be  acountable set of events of a probability space $\{\Omega, \mathcal{F}, P)$.

    Then, 

    \begin{equation}
	P\Big( \cup_{i\geq 1}A_i \Big) \leq \sum_{i\geq 1} P(A_i)
    \end{equation}
    
\end{proposition}


    This set might be infinite.

    \exercice{\textbf{}}Proove the proposition for tomorrow's practical session, in the case of a finite set of $A_i$, and optionnally for an infinite set.
    
\end{frame}

\begin{frame}{Hoeffding's inequality}

\begin{theorem}{Hoeffding's inequality}

    Let $(X_i)_{1\leq i\leq n}$ be $n$ i.i.d real random variables such that $\forall i\in [1, n]$, $X_i\in [a,b]$ and $ E(X_i)= \mu \in \mathbb{R} $.  Let $	\bar{\mu} = \frac{1}{n} \sum^{n}_{i=1} X_i$.

    Then $\forall \epsilon >0$,

    \begin{equation*}
	P\Big(| \bar{\mu}-\mu|\geq \epsilon \Big) \leq 2 \exp\Big( -\frac{2n\epsilon^2}{(b-a)^2} \Big)
    \end{equation*}
\end{theorem}

We admit this theorem.
    
\end{frame}

\subsection{Model selection}
\label{sub:model_selection}

\begin{frame}{Example}

    \begin{itemize}
        \item If $d>>n$ and we want to learn a linear model $ x \mapsto \langle \theta, x \rangle$, we have seen that this raises statistical issues (high variance, overfitting).
	\item However, if we know in advance that $ \theta$ only has $s<d$ non-zero coordinates (sparse $ \theta$), we can reformulate to an easier problem.
	\item But most of the time this is not the case, $s$ is not known, so we need to test several subsets of non-zero coordinates.
    \end{itemize}

    
\end{frame}

\begin{frame}{Example}


    We could write the following regularized optimization problem

    \begin{equation}
	\hat{ \theta} = \argmin_{ \theta\in \mathbb{R}^d }\Big( ||y- X \theta|| + \lambda || \theta||_0 \Big)
    \end{equation}

    \begin{itemize}
	\item $ y\in \mathbb{R}^n$ (labels)
	\item $X\in \mathbb{R}^{n,d}$  (design matrix)
	\item $ ||\theta||_0$: number of non-zero components of $ \theta$
    \end{itemize}

    
\end{frame}

\begin{frame}{Example}


    We could write the following regularized optimization problem

    \begin{equation}
	\hat{ \theta} = \argmin_{ \theta\in \mathbb{R}^d }\Big( ||y- X \theta|| + \lambda || \theta||_0 \Big)
    \end{equation}

    \begin{itemize}
	\item $ y\in \mathbb{R}^n$ (labels)
	\item $X\in \mathbb{R}^{n,d}$  (design matrix)
	\item $ ||\theta||_0$: number of non-zero components of $ \theta$
    \end{itemize}

    However,
    \begin{itemize}
        \item optimization issue (not convex)
	\item computationally prohibitive to test all subsets of $[1, d]$.
	    
	    \exercice{How many subsets does $[1, d]$ contain ?} 
    \end{itemize}
\end{frame}

\subsection{Lasso}%
\label{sub:lasso}



\begin{frame}{Lasso}

Le Lasso replaces $||\theta||_0$ by $ ||\theta||_1$.

\begin{equation}
    || \theta_1|| = \sum^{d}_{i=1} | \theta_i|
\end{equation}

Lasso estimator:

\begin{equation}
    \tilde{\theta}_{\lambda}\in \argmin_{\theta\in \mathbb{R}^d}\{||y-X\theta||^2+\lambda ||\theta||_1\}
\end{equation}

For technically involved reasons, the optimization with the lasso leads to sparser solutions in some situations.
    
\end{frame}

\begin{frame}{Lasso}

Lasso estimator:

\begin{equation}
    \tilde{\theta}_{\lambda}\in \argmin_{\theta\in \mathbb{R}^d}\{||y-X\theta||^2+\lambda ||\theta||_1\}
\end{equation}

For technically involved reasons, the optimization with the lasso leads to sparser solutions. Frequently used optimization algorithm:

\begin{itemize}
    \item coordinate descent (algorithm used in scikit)
    \item Fista
    \item LARS
\end{itemize}

\url{https://en.wikipedia.org/wiki/Coordinate_descent} 
    
\end{frame}

\subsection{Construction of a decision tree}%
\label{sub:construction_of_a_decision_tree}

\begin{frame}{Construction of a tree}
    
    A split node corresponds to:
    \begin{itemize}
        \item a segmentation variable $V$
	\item a segmentation value / threshold if $V$ is quantitative. If $V$ is a class, it is a two-fold partitionning of the set of classes.
    \end{itemize}
   The construction of a tree requires a criterion to:
    \begin{itemize}
        \item choose the segmentation variable and value
	\item decide when a node is terminal (leaf node)
	\item associate a prediction value to all leaf nodes
    \end{itemize}
\end{frame}

\subsubsection{Segmentation criterion}
\label{ssub:segmentation_criterion}

\begin{frame}{Homogeneity}

    We want the homogeneity in the child nodes to be greater than in the parent node. Equivalently, we can define a non-negative heterogeneity function $H$ that describes each node and that must be:
    \begin{itemize}
	\item $0$ if the node is homogeneous (only one class or one output value is represented in this node)
	\item maximal if the values of $y$ are uniformly distributed within the node.
	\item if $H(L)$ it the value of $H$ for the left child node (and $H(R)$ for the right one), then the segmentation should minimize
	    \begin{equation}
		H(L)+H(R)
	    \end{equation}
    \end{itemize}
\end{frame}

\subsubsection{Homogeneity/heterogeneity criterion}%
\label{ssub:homogeneity_criterion}

\begin{frame}{Homogeneity criterion for regression}

    In the case of regression, $ \mathcal{Y}  = \mathbb{R} $ and the heterogeneity $H(n)$ of node $n$ is its empirical variance:
    \begin{equation}
	H(n) = \frac{1}{|n|} \sum_{i\in n} (y_i - \bar{y_n})^2
    \end{equation}
    Each $y_i$ is a label for an input $x_i$ that arrives in node $n$.
    
\end{frame}

\begin{frame}{Homogeneity criterion for classification}
    In the case of classification, $ \mathcal{Y}  = [1, \dots, L]$ and several criteria exist.  
\end{frame}

\begin{frame}{Homogeneity criterion for classification: entropy (information gain)}
    \begin{itemize}
        \item We use the convention that $ 0 \log (0) = 0$
	\item $p_n^l$ is the proportion of class $ l$ in node $n$.
    \end{itemize}
    The \textbf{{entropy}} writes:

    \begin{equation}
	H(n) = - \sum^{L}_{l=1} p_n^l\log( p_n^l)
    \end{equation}

    \exercice{\textbf{}}What are the maximum and minimum values of the entropy ?
\end{frame}

\begin{frame}{Homogeneity criterion for classification: Gini impurity}
    The Gini impurity writes:

    \begin{equation}
	H(n) = \sum^{L}_{l=1} p_n^l(1-p_n^l)
    \end{equation}
\end{frame}

\begin{frame}{Homogeneity criterion for classification: Gini impurity}
    The Gini impurity writes:

    \begin{equation}
	H(n) = \sum^{L}_{l=1} p_n^l(1-p_n^l)
    \end{equation}
    \exercice{\textbf{}}Interpret the meaning of the Gini impurity in terms of probabilities, assuming that we predict according to the proportions $p_n^l$.
\end{frame}

\begin{frame}{Homogeneity criterion for classification: misclassification probability}
    \begin{equation}
	H(n) = 1-  \max_l(p_n^l)
    \end{equation}
    If we predict the most represented class in $n$, then this is the misclassification probability.
\end{frame}

\subsubsection{Prediction}%
\label{ssub:prediction}

\begin{frame}{Prediction for a leaf node}
    \begin{itemize}
        \item \textbf{{regression}}: predict the average value in the node
	\item \textbf{{classification}}: several possibilities
	    \begin{itemize}
	        \item most represented class in the node
		\item most probable class \textit{{a posteriori}} if the \textit{{a priori}} probabilities are known (Bayesian probabilities) 
		\item class that costs the less in case of missclassification
	    \end{itemize}
    \end{itemize}
\end{frame}


\subsection{Tree pruning}%
\label{sub:pruning}

\begin{frame}{Pruning}
    \begin{itemize}
	\item If the segmentation is not restricted, the tree can fit the training dataset perfectly (overfitting).
        \item learning would then be highly dependent on the choice of the training dataset, leading to \textbf{{instability.}} 
	\item To avoid it, \textbf{{pruning}} (élaguage) is used: it consists in restricting the size of the tree (and can be seen as an equivalent to regularization like in ridge regression or logistic regression). 
    \end{itemize}
\end{frame}

\begin{frame}{Pre-pruning by restricting divisions}
    We can impose
    \begin{itemize}
        \item minimum impurity decrease
        \item a minimum number of samples in a leaf node
        \item a minimum number of samples to authorize splitting a node
    \end{itemize}
\end{frame}

\subsubsection{Remarks on decision trees}%
\label{ssub:}

\begin{frame}{Remarks on decision trees I}

    \begin{itemize}
        \item Decision trees do not require hypotheses on the distribution of de features.
	\item Feature selection is integrated in the algorithm, so they might be relevant in a situation with many features (variables explicatives)
	\item A segmentation is invariant to a monotonic change in a feature. It is thus robust  to outliers or to very asymetrical distributions.
	\item Decision trees allow a straightforward \textbf{{interpretation}} of the model.
    \end{itemize}
    
\end{frame}


\begin{frame}{Remarks on decision trees II}
    \begin{itemize}
        \item Decision trees are greedy: they might find a local minimum
	\item they are hierarchical: a bad choice in a segmentation at the top of the tree propagates in the whole tree
	\item Hence their instability and sensibility to the choice of the training set.
    \end{itemize}
\end{frame}

\subsection{Bagging}%
\label{sub:bagging}


\begin{frame}{Aggregating to reduce the variance}
    \begin{itemize}
        \item usual setup: input space $ \mathcal{X} $, output space $ \mathcal{Y} $.
	\item we note $z_b$ a dataset sampled from the unknown distribution $ \rho$. We sample $b$ different datasets, $b\in [1, B]$, 
	    \begin{equation}
		z_b = \{(x_{1b}, y_{1b}), \dots (x_{nb}, y_{nb})\}
	    \end{equation}
	\item we note $ \hat{f_{z_b}}$ the estimator obtained after learning with $ z_b$.
    \end{itemize}
\end{frame}

\begin{frame}{Aggregate to reduce the variance}
    \begin{itemize}
	\item we note $z_b$ a dataset sampled from the unknown distribution $ \rho$. If we sample $b$ different datasets, $b\in [1, B]$, 
	    \begin{equation}
		z_b = \{(x_{1b}, y_{1b}), \dots (x_{nb}, y_{nb})\}
	    \end{equation}
	\item we note $ \hat{f_{z_b}}$ the estimator obtained after learning with $ z_b$.
    \end{itemize}
    \textbf{{Aggregating}} consists in using as an estimator:
    \begin{itemize}
        \item for regression
	    \begin{equation}
		\hat{f_B} = \frac{1}{B} \sum^{B}_{b=1} \hat{f_{z_b}}
	    \end{equation}
	\item for classification
	    \begin{equation}
		\hat{f_B}(x) = \argmax_{j} | \{ b, \hat{f_{z_b}}(x)=j \}|
	    \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}{Bootstrapping and bagging}
    If $B$ is large, it is not possible to sample $B$ independent datasets with $n$ samples, from a finite dataset.

    \textbf{{Bootstrapping}} consists in sampling $B$ times a sample dataset with $n$ elements \textbf{{with replacement}}.
    
    \textbf{{Bagging}} is the combination of bootstrapping and aggregating.
    
\end{frame}

\begin{frame}{Out-of-bag error}
    
    Vocabulary: for an observation $(x_i, y_i)$, the \textbf{{out-of-bag error}}  is the mean error on this observation among the estimators that were trained on a bootstrap dataset \textbf{{not}} containing it.

    It can be seen as an equivalent to the test error, the difference being that when doing bootstrapping, there is not a single train set and a single test set.
\end{frame}

\subsection{Random forest}%
\label{sub:random_forests}

\begin{frame}{Random forest}
    \begin{itemize}
        \item Random forest \cite{Breiman2001} is a bagging method with binary CART estimators, with additional randomness added to the choice of segmentation variables.
	\item the goal is to increase the \textbf{{independence}} between the base trees.
	\item although the theoretical properties of random forest are not yet fully understood, it is an important benchmark in supervised learning (but for instance not when the problem can be solved in a linear way).
    \end{itemize}
\end{frame}

\begin{frame}{Variance of an average: independent variables}

Recall that if $(X_k)_{k\in \mathbb{N} }$ is a sequence of i.i.d. real variables that have a moment of order $2$, and hence a variance $ \sigma^2$, and an expected value of $m$. Then if $S_B= \frac{1}{B} \sum^{B}_{i=1} X_i$

\begin{equation}
\begin{aligned}
    \label{eq:}
    Var(S_B)&= \sum^{B}_{i=1} Var( \frac{1}{B}  X_i)\\
    &= \frac{1}{B^2} \sum^{B}_{i=1} Var(X_i)\\
    &= \frac{\sigma^2}{B} 
\end{aligned}    
\end{equation}
    
\end{frame}

\begin{frame}{Variance of the average: correlated variables}
    However, if the $X_i$ are identically distributed but correlated with correlation $c$, then we admit that
    \begin{equation}
	Vac(S_B) = c \sigma^2 + \frac{1-c}{B} \sigma^2
    \end{equation}

    Random forest diminishes $c$ by adding some randomness in the choice of the segmentation variables.
    
\end{frame}

\begin{frame}{Random forest}

    Let $d$ be the number of features. Let $m\leq d$ be an integer.

\begin{algorithm}[H]
	\KwResult{Random forest estimator}
	\For{$b\in [1, B]$}
 	{
	    Sample a bootstrap dataset $z_b$\;
	    Estimate $ \hat{f_{z_b}}$ with \textbf{{randomization}}  of the variables. The search of the optimal segmentation is done on $m$ randomly sampled variables.\;
  	}
	    \begin{equation}
		\hat{f_B} = \frac{1}{B} \sum^{B}_{b=1} \hat{f_{z_b}}
	    \end{equation}
	
	
	return $ \hat{f_B}$
	\caption{Random forest}
\end{algorithm}
\end{frame}

\begin{frame}{Remarks on random forest}
    \begin{itemize}
	\item With random forest, it is possible to use a more brutal pruning (for instance using only trees of depth $2$)
	\item To tune $m$, heuristics are used. Common ones include:
	    \begin{itemize}
		\item $m = \sqrt{d} $ for classification ($d$ is the number of features)
		\item $m = d/3$ for regression
		\item cross validation.
	    \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Importance of the variables}
    Although the obtained estimator is hard to interpret, we can still study the statistical importance of variables for the prediction.

\begin{frame}{Mean decrease accuracy}

    Consider a variable $j$.
    For a given tree $b$
   \begin{itemize}
       \item compute the out-of-bag error.
	\item shuffle the values of $j$ in the out-of-bag sample.
	\item compute the new out-of-bag error
	\item store the decrease in prediction quality, $D_j^b$.
   \end{itemize}

\subsection{Boosting}%
\label{sub:boosting}

\begin{frame}{Boosting}
    \begin{itemize}
        \item While bagging is a random method, \textbf{{boosting}} is an adaptive method.
	\item Boosting builds on \textit{{weak classifiers}}, and like bagging, aggregates them, for instance with a \textbf{{weighted sum}}.
	\item However, the weak classifiers are built \textbf{{sequentially}}. The classifier $b+1$ adapts the classifier $b$ by focusing on improving the prediction of incorrectly classified training samples.
	\item Several variants exists (in the weighting, loss function, aggregating method, etc).
    \end{itemize}
\end{frame}

\begin{frame}{Adaboost}
    \begin{itemize}
	\item  Original boosting algorithm \cite{Freund1996} 
	\item $ \mathcal{Y}  = \{-1, 1\}$ (but can also be adapted to a regression problem)
	\item  Given the sample dataset $z =  \{(x_{1}, y_{1}), \dots (x_{n}, y_{n})\}$. We learn a \textbf{{sequence of estimators}} $(\delta_m)_{m\in [1..B]}$ (often CART).
    \end{itemize}
\end{frame}

\begin{frame}

\begin{algorithm}[H]
\Fontvi
	\text{Initialize the weights} $w = \{w_i=\frac{1}{n}, i=1..n  \}$\;
	\For{$m=1..B$}
 	{
	    Estimate $\delta_m$ on the weighted dataset.
	    Compute the error rate
	    \begin{equation}
		\hat{\epsilon_m} = \frac{ \sum^{n}_{i=1} w_i 1( \delta_m(x_i)\neq y_i)}{ \sum^{n}_{i=1} w_i}
	    \end{equation}
	    Compute the logit of $\hat{\epsilon_m}$
	    \begin{equation}
		c_m = \log \big( \frac{1- \hat{\epsilon_m}}{ \hat{\epsilon_m}} \big) 
	    \end{equation}
	    Update the weights
	    \begin{equation}
		w_i \leftarrow w_i \exp\Big(c_m 1( \delta_m(x_i)\neq y_i) \Big)
	    \end{equation}
  	}
	    \begin{equation}
		\hat{\delta_B} = \text{sign} \Big(\sum^{B}_{b=1} \delta_{z_b}\Big)
	    \end{equation}
	
	
	return $ \hat{f_B}$
	\caption{Adaboost for binary classification (adaptive boosting)}
\end{algorithm}
    
\end{frame}

\begin{frame}{Remarks}

    We need to enforce that $c_m \geq 0$. It is verified if $\hat{\epsilon_m} \leq \frac{1}{2} $.

    \begin{equation}
        c_m = \log \big( \frac{1- \hat{\epsilon_m}}{ \hat{\epsilon_m}} \big) 
    \end{equation}
    
\end{frame}

\subsubsection{Gradient tree boosting}%
\label{ssub:e}

\begin{frame}{Gradient tree boosting}
    \begin{itemize}
	\item differentiate with respect to the predictions of the tree (compute a gradient)
	\item approximate the gradient by a regression tree
	\item update the tree following this gradient.
    \end{itemize}
\end{frame}

\begin{frame}{Gradient tree boosting}
\begin{algorithm}[H]
\Fontvi
	\text{Initialize} $f_0 = \argmin_{z} \sum^{n}_{i=1} l(y_i, z)$\;
	\For{$m=1..B$}
 	{
	    Compute $r_{m_i} = - \frac{\partial l (y_i, f(x_i))}{\partial f(x_i)}(f=f_{m-1})  , i\in [1, n]$

	    Train a decision tree $ \delta_m$ on $(x_i, r_{m_i})_{i\in [1, n]}$

	    Compute $ \gamma_m$ minimizing

	    \begin{equation}
		\gamma \rightarrow \sum^{n}_{i=1} l(y_i, f_{m-1}(x_i) + \gamma \delta_m(x_i))
	    \end{equation}

	    Update the estimator
	    \begin{equation}
		\hat{f_m} = \hat{f_{m-1}}+ \gamma_m \delta_m
	    \end{equation}
  	}
	return $ \hat{f_B}$
	\caption{Gradient tree boosting}
\end{algorithm}
\end{frame}

\begin{frame}{Gradient tree boosting}

	\item shrinkage $\eta$ ($ \hat{f_m} = \hat{f_{m-1}}+ \eta\gamma_m \delta_m$, with $0\leq \eta \leq 1$). If $ \eta$ is low, e.g. $\leq 0.1$, it can lead to a slower convergence but might prevent overfitting.
	\item number of trees learned $B$
    \end{itemize}

    \url{https://www.analyticsvidhya.com/blog/2016/02/complete-guide-parameter-tuning-gradient-boosting-gbm-python/} 

    It is necessary to experiment, read documentations, cross validate, use heuristics, etc.
