\exercice{\textbf{}}

\begin{align}
    \label{eq:}
    ||x^*||^2&=\sum_{i=1}^{d}(x_i^*)^2\\
    &=\sum_{i=1}^{2t+1}(x_i^*)^2\\
    &=\sum_{i=1}^{2t+1}(1- \frac{i}{2t+2} )^2\\
    &=\sum_{i=1}^{2t+1}(\frac{2t+2-i}{2t+2} )^2\\
    &=\sum_{j=1}^{2t+1}(\frac{j}{2t+2} )^2\\
    &=(\frac{1}{2t+2})^2\sum_{j=1}^{2t+1}j^2\\
    &=(\frac{1}{2t+2})^2(2t+1)(2t+2)(4t+3)/6\\
    &=\frac{1}{2t+2}(2t+1)(4t+3)/6\\
    &\leq(4t+3)/6\\
    &\leq(4t+4)/6\\
    &=(2t+2)/3\\
\end{align}
